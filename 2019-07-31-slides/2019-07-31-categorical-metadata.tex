\section{Proposal}
\begin{frame}
    \begin{enumerate}
        \item Gather images at a \textbf{single resource}.
        \item Establish a \textbf{common description framework} for image metadata.
        \item Provide \textbf{bulk, programmatic access} to image subsets.
    \end{enumerate}
\end{frame}

\section{Functions}

\begin{frame}

    \end{enumerate}
\end{frame}
